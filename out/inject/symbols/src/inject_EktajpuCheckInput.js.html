 93     /**
 94      * Checks text from HTML element.
 95      * @param sentence String to convert.
 96      * @return String Converted text.
 97      */
 98     EktajpuCheckInput.prototype.checkSentence = function (sentence) {
 99         var numberOfChanges = 0;
100         // cycle through string from input box
101         var valTextOrig = sentence;
102         var valTextNew = "";
103         var posCharEnd = 0;
104         var skip = false;
105         for (var pos = 0; pos < valTextOrig.length; pos++) {
106             if (skip == true && pos <= posCharEnd) {
107                 continue;
108             }
109             else {
110                 skip = false;
111             }
112             var valTextTemp = "";
113             // loop through character list
114             for (var i = 0; i < this.charactersConvertFrom.length; i++) {
115                 posCharEnd = (pos + this.charactersConvertFrom[i].length) - 1;
116                 var charSet = valTextOrig.substr(pos, this.charactersConvertFrom[i].length);
117                 if (charSet == this.charactersConvertFrom[i]) {
118                     skip = true;
119                     numberOfChanges++;
120                     valTextNew += this.charactersConvertTo[i];
121                     break;
122                 }
123             }
124             if (skip == true)
125                 continue;
126             // add char to new string
127             valTextNew += valTextOrig[pos];
128         }
129         // set new text on gui
130         if (numberOfChanges > 0) {
131             return valTextNew;
132         }
